23 internal static ScenarioBuilderEditorConfig FindConfig()
24 {
25 var asset = AssetDatabase.FindAssets($"t:{nameof(ScenarioBuilderEditorConfig)}")
26 .Select(AssetDatabase.GUIDToAssetPath)
27 .Select(AssetDatabase.LoadAssetAtPath<ScenarioBuilderEditorConfig>)
28 .FirstOrDefault(x => x != null);
29 if(asset == null)
30 {
31 asset = CreateInstance<ScenarioBuilderEditorConfig>();
32
33 string folderPath = "Assets/BX/Editor/BxUni-ScenarioBuilder";
34
35 var split = folderPath.Split('/');
36 var sb = new StringBuilder(split[0]);
37 foreach (string folder in split.Skip(1))
38 {
39 if (!AssetDatabase.IsValidFolder($"{sb}/{folder}"))
40 {
41 AssetDatabase.CreateFolder(sb.ToString(), folder);
42 sb.Append($"/{folder}");
43 }
44 }
45
46 string path = $"{folderPath}/{nameof(ScenarioBuilderEditorConfig)}.asset";
47 AssetDatabase.CreateAsset(asset, path);
48 AssetDatabase.SaveAssets();
49 AssetDatabase.Refresh();
50 }
51
52 return asset;
53 }
